(i)\\
Mit \# gekennzeichnete lines bezeichnen Kommentare.\\
Das Editieren der Datei, musste nicht angegeben werden, deshalb ist es 
    auskommentiert.\\
\begin{verbatim}
    # S1:
    $ git init
    # touch data.txt
    $ git add data.txt
    $ git commit -m "first commit"
    # S2:
    $ git checkout -b fork
    # S3:
    # echo f > data.txt
    $ git  add data.txt
    $ git commit -m "fork commit"
    $ git checkout master
    # echo m > data.tx
    $ git add data.txt
    $ git commit -m "master commit"
    # S4:
    $ git merge fork
    # An dieser stelle werden wir auf einen merge-conflict aufmerksam gemacht
    $ git mergetool
    $ git commit -m "resolved merge-conflict"
\end{verbatim}
